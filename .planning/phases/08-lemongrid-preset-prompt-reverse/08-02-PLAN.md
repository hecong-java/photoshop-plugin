    function extractInputValues(params: Record<string, unknown>): Record<string, string | number | boolean> {

        if (typeof value === 'string' || typeof value === 'number' || typeof value === 'boolean') {

        if (typeof value === 'string' && value.length > 10 && !isNaN(Date.parse(value)) === false && key.toLowerCase().includes('image')) {
          result[key] = value;
        }
      }
      return result;
    }
