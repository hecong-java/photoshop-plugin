
    Add a new `settings-card dashscope-config` div INSIDE the `settings-grid` div, AFTER the `comfy-connection` card and BEFORE the `capabilities-matrix` card:

    ```tsx
          {dashScope.apiKey ? '已配置' : '未配置'}

      </div>

      <div className="connection-form">
        <div className="form-group">

          />
        </div>
        <div className="form-group">
          <label htmlFor="dashscope-model">模型</label>
    ```
