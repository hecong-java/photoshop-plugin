**LemonGridClient class:**
```typescript
export class LemonGridClient {
  private serverUrl: string;

  constructor(options: { serverUrl: string }) {
    this.serverUrl = options.serverUrl.replace(/\/+$/, '');
  }

  private async fetchWithAuth(path: string, options?: RequestInit): Promise<Response> {
    await ensureValidToken();
    const token = useLemonGridStore.getState().accessToken;
    const url = `${this.serverUrl}${path}`;
    const headers: Record<string, string> = {
      ...(options?.headers as Record<string, string> || {}),
    };
    if (token) {
      headers['Authorization'] = `Bearer ${token}`;
    }
    return lemongridFetch(url, { ...options, headers });
  }

  private async fetchJson<T>(path: string, options?: RequestInit): Promise<T> {
    const response = await this.fetchWithAuth(path, {
      ...options,
      headers: { 'Content-Type': 'application/json', ...(options?.headers as Record<string, string> || {}) },
    });
    if (!response.ok) {
      const errorBody = await response.text();
      throw new Error(`LemonGrid API error ${response.status}: ${errorBody.substring(0, 200)}`);
    }
    return response.json() as Promise<T>;
  }
}
```

**handleClusterGenerate function:**
```typescript
const handleClusterGenerate = async () => {
  setIsGenerating(true);
  try {
    const client = new LemonGridClient({ serverUrl: lemonGridServerUrl });
    // Per D-41: snapshot parameter values at submit time
    const snapshotParams = { ...templateParams };
    const result = await client.submitTask(selectedTemplate.id, snapshotParams);
    // Add task to lemongridStore for tracking
    useLemonGridStore.getState().updateTask(result.id, {
      taskId: result.id,
      templateId: selectedTemplate.id,
      templateName: selectedTemplate.name,
      status: result.status as any,
      progress: 0,
      progressDetail: null,
      queuePosition: null,
      errorCode: null,
      errorMessage: null,
      outputAssetIds: [],
      submittedAt: Date.now(),
      completedAt: null,
      durationSeconds: null,
      params: snapshotParams,
      thumbnail: null,
    });
    // Per D-47: auto-display results when task completes (polling is in Plan 06-03)
    // Start polling for this task
    startTaskPolling(result.id);
  } catch (error) {
    // Per D-45: show user-friendly error
    const message = error instanceof Error ? error.message : '任务提交失败';
    setProgress(prev => ({ ...prev, status: 'error', error: message }));
    setIsGenerating(false);
  }
};
```

```typescript
const startTaskPolling = async (taskId: string) => {
  const client = new LemonGridClient({ serverUrl: lemonGridServerUrl });
  const poll = async () => {
    try {
      await ensureValidToken(); // Per D-42: auto re-auth on 401
      const status = await client.getTaskStatus(taskId);
      useLemonGridStore.getState().updateTask(taskId, {
        status: status.status,
        progress: status.progress,
        progressDetail: status.progress_detail,
        queuePosition: status.queue_position,
        errorCode: status.error_code,
        errorMessage: status.error_message,
        outputAssetIds: status.output_file_ids || [],
        completedAt: status.completed_at ? new Date(status.completed_at).getTime() : null,
        durationSeconds: status.duration_seconds,
      });
      // Update progress bar for the current task
      if (['PENDING', 'QUEUED', 'SYNCING', 'RUNNING'].includes(status.status)) {
        // Per D-28: adaptive interval
        const interval = status.status === 'RUNNING' ? 1000 : 2000;
        setTimeout(poll, interval);
      } else {
        // Task finished
        setIsGenerating(false);
        if (status.status === 'COMPLETED') {
          await downloadAndImportResults(taskId, status.output_file_ids);
        }
      }
    } catch (error) {
      // Per D-31: keep polling until completion/failure/cancel
      setTimeout(poll, 2000);
    }
  };
  poll();
};
```

```typescript
const downloadAndImportResults = async (taskId: string, assetIds: string[]) => {
  const client = new LemonGridClient({ serverUrl: lemonGridServerUrl });
  const images: Array<{ url: string; blob: Blob; filename: string; assetId: string }> = [];
  for (const assetId of assetIds) {
    try {
      const blob = await client.downloadAsset(assetId);
      const filename = `cluster-${assetId.substring(0, 8)}.png`;
      images.push({ url: URL.createObjectURL(blob), blob, filename, assetId });
      // Per D-35: auto-import to PS as separate layer
      await syncGeneratedImageToPsLayer(blob, filename);
    } catch (e) {
      console.error('[Draw] Failed to download cluster result:', assetId, e);
    }
  }
  // Per D-51: independent clusterOutputImages
  images.forEach(img => useLemonGridStore.getState().addClusterOutputImage(img));
  // Per D-47: auto-display results
  setOutputImages(images.map(img => img.url)); // Reuse existing output display
  setIsGenerating(false);
};
```
